def bootstrap(data, K=1000):
  dataVec = np.zeros(K)
  for k in range(K):
    dataVec[k] = np.average(np.random.choice(data, len(data)))
  Avg = np.average(dataVec)
  Var = np.var(dataVec)
  Std = np.std(dataVec)
    
  return Avg, Var, Std
